// new Set();
1. key 없이 값만 가지는 저장 방식
2. 중복을 허용하지 않는 저장 방식


let st = new Set(["김승겸","김선향","송은선","신상수","임민지","김선향"]);
    st.add("안태균"); // 넣기
    st.delete("송은선"); // 빼기
    st.has("신상수"); // 들어있나 불리언값
    st.size; // 길이
    st.clear(); // 다 비우기


// Set을 배열로 바꾸기
=> Array.from(Set이름);

       Set을 배열로 바꾸는 방법을 익혀두는 것이 바람직
